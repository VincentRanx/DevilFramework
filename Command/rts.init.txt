void ImHere (a, b)
{
	如果 a 大于 b 就 logi(a," 是大于 ",b," 的");
	否则 logi(a," 不大于 ",b);
};

void loadScene(sceneName)
{
	if (typeof(sceneName) == int) then
	{
		strongInvoke(global SceneManager, "LoadScene", sceneName);
	}
	else
	{
		strongInvoke(global SceneManager, "LoadScene", toString(sceneName));
	};
};
